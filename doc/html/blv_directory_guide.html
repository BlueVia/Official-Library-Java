Directory API guide

Abstract

The Bluevia Directory API is a set of functions which allows users to retrieve user-related information, divided in three blocks (user profile, access information and user terminal information). Groups of blocks can be obtained using the UserInfo class, or you can just retrieve one block using the corresponding object. This guide provides some code samples to develop applications using the Bluevia Directory functionality.

User Info Example

OAuthToken consumer = new OAuthToken("consumer_key", "consumer_secret");
OAuthToken accesstoken = new OAuthToken("token", "token_secret");

UserInfo ui = new UserInfo(consumer, accesstoken, Mode.LIVE);

UserInfoType uit = ui.get();

if (uit != null) {
                System.out.println(" --- UserInfoType --- ");
                if (uit.getUserAccessInfo() != null) {
                    System.out.println("UserAccessInfo");
                    System.out.println("     Apn:" + uit.getUserAccessInfo().getValue().getApn().getValue());
                }
                if (uit.getUserProfile() != null) {
                    System.out.println("UserProfile");i
                    // do stuff
                }
                if (uit.getUserTerminalInfo() != null) {
                    System.out.println("UserTerminalInfo");
                    // do stuff
                }
            }

User Profile Example

OAuthToken consumer = new OAuthToken("consumer_key", "consumer_secret");
OAuthToken accesstoken = new OAuthToken("token", "token_secret");

Profile ui = new Profile(consumer, accesstoken, Mode.LIVE);

ProfileFields[] dataSets = new ProfileFields[] { ProfileFields.parentalControl, ProfileFields.operatorId };

UserProfileType utt = ui.get(dataSets);

if (utt != null) {
    System.out.println(" --- UserProfileType --- ");

    System.out.println("     Ocb:" + utt.getOcb());
    System.out.println("     OperatorId:" + utt.getOperatorId());

}
